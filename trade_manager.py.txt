import time
from typing import Dict, List

PAPER_ORDERS = []

def place_paper_order(symbol: str, side: str, price: float, size: float, note: str=""):
    order = {
        "ts": time.time(),
        "symbol": symbol,
        "side": side,
        "price": price,
        "size": size,
        "note": note
    }
    PAPER_ORDERS.append(order)
    return order

def list_paper_orders():
    return PAPER_ORDERS
